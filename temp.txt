#include<iostream>
using namespace std;
int main(){
    int x,n,q,l,r;
    string messageVar;
    char temp;
    string data="abcdefghijklmnopqrstuvwxyz";
    cin>>x;
    while(x>0){
        x--;
        cin>>n>>q;
        getline(cin, messageVar);
        cout<<"reached here1"<<endl;
        while(q>0){
            cout<<"reached here"<<endl;
            cin>>l>>r;
            l--;r--;q--;
            for(int b=0;b<=n;b++){
                if(b>=l&&b<=r){
                    temp=messageVar[b];
                    cout<<temp<<endl;
                    for(int a=0;a<data.length();a++)
                    {
                        if(data[a]==temp){
                            if((a+1)>25)
                                temp=data[0];
                            else
                                temp=data[a+1];
                            messageVar[b]=temp;
                        }
                    }
                    cout<<temp<<endl;
                }
            }
            cout<<messageVar<<endl;
        }
    }    
    return 0;
}
